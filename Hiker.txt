local ROOM_MATERIAL=Enum.Material.Slate
local ROOM_COLOR=Color3.fromRGB(70,70,70)
local DOOR_MATERIAL=Enum.Material.Cobblestone
local DOOR_COLOR=Color3.fromRGB(55,55,55)
local LOCK_COLOR=Color3.fromRGB(95,95,95)
local BLACKLIST_NAMES={figure=true,seek=true,screech=true,dupe=true,item=true,pickup=true,books=true,entity=true,monster=true}
local function isBlacklisted(p)
local a=p
while a do
if a:IsA("Model")or a:IsA("Folder")then
local n=a.Name:lower()
for b in pairs(BLACKLIST_NAMES)do
if n:find(b)then return true end
end
end
a=a.Parent
end
return false
end
local function isDoorPart(p)
local m=p:FindFirstAncestorWhichIsA("Model")
return m and m.Name:lower():find("door")
end
local function isLockPart(p)
return p.Name:lower():find("lock")
end
local function stoneifyPart(p)
if not p:IsA("BasePart")then return end
if isBlacklisted(p)then return end
if isDoorPart(p)then
p.Material=DOOR_MATERIAL
p.Color=isLockPart(p)and LOCK_COLOR or DOOR_COLOR
else
p.Material=ROOM_MATERIAL
p.Color=ROOM_COLOR
end
p.Reflectance=0.05
p.CastShadow=true
end
local function stoneifyRooms()
local rooms=Workspace:FindFirstChild("CurrentRooms")
if not rooms then return end
for _,r in ipairs(rooms:GetChildren())do
for _,p in ipairs(r:GetDescendants())do
stoneifyPart(p)
end
end
end
stoneifyRooms()
if Workspace:FindFirstChild("CurrentRooms")then
Workspace.CurrentRooms.ChildAdded:Connect(function(r)
r.DescendantAdded:Connect(stoneifyPart)
stoneifyRooms()
end)
end
